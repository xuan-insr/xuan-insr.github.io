#### 2 写汇编
```c
void main(char *s,int *n){
     char c,ch;
     c = '3';
     ch = '5';
     *n = replace(*s,c,ch);
}

int replace(char *u,char c,char ch){
     int i = 0;
     while(u[i]!=0){
          if(u[i]==c){
                u[i]=ch;
                break;
          }
          i++;
     }
     return i;
}
```
