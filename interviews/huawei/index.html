    class Solution {
public:
    int sum_k(vector<int> &v, int begin, int k) {
        int n = v.size();
        int sum = 0;
        for (int i = 0; i < k; i++)
            sum += v[(begin + i) % n];
        return sum;
    }

    vector<int> decrypt(vector<int>& code, int k) {
        if (k == 0) return vector<int>(code.size(), 0);

        int n = code.size();
        vector<int> result(code.size());

        if (k > 0) {
            for (int i = 0; i < n; i++)
                result[i] = sum_k(code, (i + 1) % n, k);
            return result;
        }

        for (int i = 0; i < n; i++)
            result[i] = sum_k(code, (i + n + k) % n, -k);
        return result;
    }
};
